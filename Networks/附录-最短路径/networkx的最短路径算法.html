# version 1.0
from tqdm import tqdm
import networkx
import numpy as np

def networkx_dijkstra(m, source=None, target=None):
    G = networkx.Graph()
    for i in m.columns:
        G.add_node(i)
    for i in range(len(m.columns)):
        for j in range(i+1, len(m.columns)):
            if not np.isnan(m.iloc[i,j]):
                G.add_edge(m.columns[i], m.columns[j], weight=m.iloc[i,j])
    
    if source is not None and target is not None:
        return nx.dijkstra_path_length(G, source=source, target=target)
    
    p = m*np.nan
    for i in range(len(m.columns)):
        for j in range(i, len(m.columns)):
            try:
                p.iloc[i, j] = networkx.dijkstra_path_length(G, source=m.columns[i], target=m.columns[j])
            except networkx.exception.NetworkXNoPath:
                p.iloc[i, j] = np.nan
            p.iloc[j, i] = p.iloc[i, j]
    return p
